# npm

```
#初始化
npm init

#局部安装
npm install lodash

#局部更新
npm update
npm outdated

#局部卸载
npm uninstall lodash

#全局安装
npm install -g jshint

#全局更新
npm update -g jshint
npm update -g

#查看哪些全局安装的包需要更新
npm outdated -g --depth=0

#全局卸载
npm uninstall -g jshint

#查看所有全局安装
npm list -g

#查看某个模块的版本号
npm list grunt

#搜索模块
npm search express
```
